    import type React from "react";
    import {
      GithubLogo,
      LinkedinLogo,
     Envelope,
    } from "@hugeicons/react";

    type SocialLink = {
      name: string;
      href: string;
      icon: React.ComponentType<{ className?: string }>;
    };

    const socialLinks: SocialLink[] = [
      {
        name: "GitHub",
        href: "https://github.com/florian",
        icon: GithubLogo,
      },
      {
        name: "LinkedIn",
        href: "https://linkedin.com/in/florian",
        icon: LinkedinLogo,
      },
      {
        name: "Email",
        href: "mailto:hello@florian.dev",
        icon: Envelope,
      },
    ];

    type SocialLinksProps = {
      className?: string;
    };

    export const SocialLinks: React.FC<SocialLinksProps> = ({ className = "" }) => {
      return (
        <div className={`flex gap-4 ${className}`}>
          {socialLinks.map((link) => {
            const Icon = link.icon;
            return (
              <a
                key={link.name}
                href={link.href}
                target="_blank"
                rel="noopener noreferrer"
                aria-label={link.name}
                className="text-foreground hover:text-primary transition-colors"
              >
                <Icon className="size-5" />
              </a>
            );
          })}
        </div>
      );
    };

    Uses HugeIcons which is already installed from Phase 1.
    Icons only, no labels (per CONTEXT.md requirements).
  </action>
  <verify>file exists and exports SocialLinks component</verify>
